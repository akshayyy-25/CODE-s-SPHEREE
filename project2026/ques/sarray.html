using namespace std;

cout<<"enter size of 1st array: ";
cin>>x;
cout<<"enter size of 2nd array: ";
cin>>y;
int arr[x];
int arr2[y];
int arr3[x+y];
for(int i=0;i<x;i++){
    cout<<"enter elements of 1st array: ";
    cin>>arr[i];
}
for(int i=0;i<y;i++){
    cout<<"enter elements of 2nd array: ";
    cin>>arr2[i];
}
int i=0,j=0,k=0;
while(i<x && j<y){
    if(arr[i]<arr2[j]){
        arr3[k]=arr[i];
        i++;
    }
    else{
        arr3[k]=arr2[j];
        j++;
    }
    k++;
}
while(i<x){
    arr3[k]=arr[i];
    i++;
    k++;
}
while(j<y){
    arr3[k]=arr2[j];
    j++;
    k++;
}
cout<<"merged array: ";
for(int i=0;i<x+y;i++){
    cout<<arr3[i]<<" ";
}
return 0;

    cout << "Enter " << n << " elements:\n";

        cin >> arr[i];
