  1. Program to find all subsets of a string 

                
#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main() {
    string str;
    cout << "Enter a string: ";
    cin >> str;

    int n = str.length();
    int total = pow(2, n);

    cout << "All subsets are:\n";

    for (int i = 0; i < total; i++) {
        string subset = "";

        for (int j = 0; j < n; j++) {
            // Check if jth bit is set
            if (i & (1 << j)) {
                subset += str[j];
            }
        }

        cout << subset << endl;
    }

    return 0;
}

  1. Program to find all the permutations of a string 

                
#include <iostream>
#include <string>
using namespace std;

void permute(string str, int left, int right) {
    if (left == right) {
        cout << str << endl;
    } else {
        for (int i = left; i <= right; i++) {
            
            swap(str[left], str[i]);

            permute(str, left + 1, right);

            
            swap(str[left], str[i]);
        }
    }
}

int main() {
    string str;
    cout << "Enter a string: ";
    cin >> str;

    cout << "All permutations are:\n";
    permute(str, 0, str.length() - 1);

    return 0;
}

  1. Program to find maximum and minimum occurring character in a string 

                
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    cout << "Enter a string: ";
    getline(cin, str);

    int freq[256] = {0};

    for (int i = 0; i < str.length(); i++) {
        freq[str[i]]++;
    }

    char maxChar, minChar;
    int max = 0, min = str.length();

    for (int i = 0; i < str.length(); i++) {
      
        if (str[i] == ' ')
            continue;

        if (freq[str[i]] > max) {
            max = freq[str[i]];
            maxChar = str[i];
        }

        if (freq[str[i]] < min && freq[str[i]] != 0) {
            min = freq[str[i]];
            minChar = str[i];
        }
    }

    cout << "Maximum occurring character: " << maxChar 
         << " (Frequency: " << max << ")" << endl;

    cout << "Minimum occurring character: " << minChar 
         << " (Frequency: " << min << ")" << endl;

    return 0;
}

  1. Program to find the largest and smallest word in a string 

                
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str[200];
    cout << "Enter a string: ";
    cin.getline(str, 200);

    int i = 0, len = 0;
    int minLen = 1000, maxLen = 0;
    char word[50], smallest[50], largest[50];

    while (true)
     {
        
        if (str[i] == ' ' || str[i] == '\0') {
            word[len] = '\0';

            if (len > 0) {
                
                if (len < minLen) {
                    minLen = len;
                    strcpy(smallest, word);
                }

                if (len > maxLen) {
                    maxLen = len;
                    strcpy(largest, word);
                }
            }

            len = 0; 

            if (str[i] == '\0')
                break;
        } 
        else 
        {
            word[len++] = str[i];
        }
        i++;
    }

    cout << "\nSmallest word: " << smallest;
    cout << "\nLargest word: " << largest;

    return 0;
}

  1. Program to find the longest repeating sequence in a string 

                
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str[100];
    cout << "Enter a string: ";
    cin.getline(str, 100);

    int n = strlen(str);
    int maxLen = 0;
    char longest[100] = "";

    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            int k = 0;

            while (str[i + k] == str[j + k] && (i + k) < n && (j + k) < n) {
                k++;
            }

            if (k > maxLen) {
                maxLen = k;
                strncpy(longest, str + i, k);
                longest[k] = '\0';
            }
        }
    }

    if (maxLen > 0)
        cout << "\nLongest Repeating Sequence: " << longest;
    else
        cout << "\nNo repeating sequence found.";

    return 0;
}

  1. Program to find the most repeated word in a text file 

                
#include <iostream>
#include <fstream>
#include <map>
using namespace std;

int main() {
    ifstream file("data.txt");  
    string word;

    map<string, int> freq;  

    while (file >> word) 
    {
        freq[word]++;
    }

    file.close();

    string maxWord;
    int maxCount = 0;

    for (auto it : freq) {
        if (it.second > maxCount) {
            maxCount = it.second;
            maxWord = it.first;
        }
    }

    cout << "Most repeated word: " << maxWord << endl;
    cout << "Frequency: " << maxCount << endl;

    return 0;
}
