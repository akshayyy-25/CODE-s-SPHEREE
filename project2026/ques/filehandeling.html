  1. Write a C++ program to store the information of 10 Students having rollno,name,marks,mobileno properties, get() and set() member functions as binary file with following options 1. Insert Record 2. Delete Record 3. Update Record (based on rollno) 4. Search Record (based on rollno) 5. Clone the file 6. Exit. 

                
#include <iostream>
#include <fstream>
using namespace std;

class Student {
    int rollno;
    string name;
    float marks;
    string mobileno;

public:
    void get() {
        cout << "Enter Roll No: ";
        cin >> rollno;
        cin.ignore();

        cout << "Enter Name: ";
        getline(cin, name);

        cout << "Enter Marks: ";
        cin >> marks;

        cout << "Enter Mobile No: ";
        cin >> mobileno;
    }

    void display() {
        cout << "\nRoll No: " << rollno;
        cout << "\nName: " << name;
        cout << "\nMarks: " << marks;
        cout << "\nMobile No: " << mobileno << endl;
    }

    int getRollNo() {
        return rollno;
    }

    // Write object to binary file
    void writeToFile(ofstream &out) {
        size_t len;

        out.write((char*)&rollno, sizeof(rollno));

        len = name.size();
        out.write((char*)&len, sizeof(len));
        out.write(name.c_str(), len);

        out.write((char*)&marks, sizeof(marks));

        len = mobileno.size();
        out.write((char*)&len, sizeof(len));
        out.write(mobileno.c_str(), len);
    }

    // Read object from binary file
    void readFromFile(ifstream &in) {
        size_t len;
        char *buffer;

        in.read((char*)&rollno, sizeof(rollno));

        in.read((char*)&len, sizeof(len));
        buffer = new char[len + 1];
        in.read(buffer, len);
        buffer[len] = '\0';
        name = buffer;
        delete[] buffer;

        in.read((char*)&marks, sizeof(marks));

        in.read((char*)&len, sizeof(len));
        buffer = new char[len + 1];
        in.read(buffer, len);
        buffer[len] = '\0';
        mobileno = buffer;
        delete[] buffer;
    }
};

// Insert Record
void insertRecord() {
    ofstream file("students.dat", ios::binary | ios::app);
    Student s;

    s.get();
    s.writeToFile(file);

    file.close();
    cout << "Record Inserted!\n";
}

// Display All
void displayAll() {
    ifstream file("students.dat", ios::binary);
    Student s;

    while (file.peek() != EOF) {
        s.readFromFile(file);
        s.display();
    }

    file.close();
}

// Search
void searchRecord() {
    ifstream file("students.dat", ios::binary);
    Student s;
    int roll;
    bool found = false;

    cout << "Enter Roll No: ";
    cin >> roll;

    while (file.peek() != EOF) {
        s.readFromFile(file);
        if (s.getRollNo() == roll) {
            s.display();
            found = true;
            break;
        }
    }

    if (!found)
        cout << "Record Not Found!\n";

    file.close();
}

// Delete
void deleteRecord() {
    ifstream file("students.dat", ios::binary);
    ofstream temp("temp.dat", ios::binary);

    Student s;
    int roll;
    bool found = false;

    cout << "Enter Roll No to Delete: ";
    cin >> roll;

    while (file.peek() != EOF) {
        s.readFromFile(file);
        if (s.getRollNo() != roll) {
            s.writeToFile(temp);
        } else {
            found = true;
        }
    }

    file.close();
    temp.close();

    remove("students.dat");
    rename("temp.dat", "students.dat");

    if (found)
        cout << "Deleted Successfully!\n";
    else
        cout << "Record Not Found!\n";
}

// Update
void updateRecord() {
    ifstream file("students.dat", ios::binary);
    ofstream temp("temp.dat", ios::binary);

    Student s;
    int roll;
    bool found = false;

    cout << "Enter Roll No to Update: ";
    cin >> roll;

    while (file.peek() != EOF) {
        s.readFromFile(file);
        if (s.getRollNo() == roll) {
            cout << "Enter New Data:\n";
            s.get();
            found = true;
        }
        s.writeToFile(temp);
    }

    file.close();
    temp.close();

    remove("students.dat");
    rename("temp.dat", "students.dat");

    if (found)
        cout << "Updated Successfully!\n";
    else
        cout << "Record Not Found!\n";
}

// Clone File
void cloneFile() {
    ifstream src("students.dat", ios::binary);
    ofstream dest("clone.dat", ios::binary);

    dest << src.rdbuf();

    cout << "File Cloned!\n";

    src.close();
    dest.close();
}

// Main
int main() {
    int choice;

    do {
        cout << "\n===== MENU =====\n";
        cout << "1. Insert\n2. Delete\n3. Update\n4. Search\n5. Clone\n6. Display\n7. Exit\n";
        cout << "Enter Choice: ";
        cin >> choice;

        switch (choice) {
            case 1: insertRecord(); break;
            case 2: deleteRecord(); break;
            case 3: updateRecord(); break;
            case 4: searchRecord(); break;
            case 5: cloneFile(); break;
            case 6: displayAll(); break;
            case 7: cout << "Exit...\n"; break;
            default: cout << "Invalid!\n";
        }

    } while (choice != 7);

    return 0;
}
